Example usage of 3B-dotiw.js

Distance of time in words

threeB.DOTIW.distance_of_time_in_words(0, 500);
// Result:

threeB.DOTIW.distance_of_time_in_words(0, 800);
// Result:

threeB.DOTIW.distance_of_time_in_words(0, 50000);
// Result:

threeB.DOTIW.distance_of_time_in_words(0, 500);
// Result:

threeB.DOTIW.distance_of_time_in_words(0, 500);
// Result:

threeB.DOTIW.distance_of_time_in_words(0, 3463212, true);
// Result:

threeB.DOTIW.distance_of_time_in_words(0, 3463212, true, {vague: true});
// Result:

threeB.DOTIW.distance_of_time_in_words(0, 3463212, {locale: {'x_hours': {'other': 'some hours'}}}));
// Result:

threeB.DOTIW.distance_of_time_in_words(0, 3463212, true, {locale: {'x_hours': {'other': 'some hours'}}}));
// Result:

threeB.DOTIW.distance_of_time_in_words(0, 3463212, true, {
  locale: {'x_hours':
    {'lots': 'a whole lot of hours',
     'pluralization': function(i) { return 'lots'; }
    }
  }
}));
// Result:
